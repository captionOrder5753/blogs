    import Idb from "idb.js";

export function getDb(config) {
  return Idb(config);
}

export function setItem(key, value = {}, store = sessionStorage) {
  if (store.db) {
    return new Promise((resolve, reject) => {
      try {
        store.insert({
          tableName: key,
          data: value,
          success: () => resolve(true),
        });
      } catch (error) {
        console.log(error);
        reject(false);
      }
    });
  } else {
    return new Promise((resolve, reject) => {
      try {
        store.setItem(key, JSON.stringify(value));
      } catch (error) {
        console.log(error);
        reject(false);
      }
    });
  }
}

export function getItem(key, store = sessionStorage, condition) {
  if (store.db) {
    return new Promise((resolve, reject) => {
      try {
        store.query_by_index({
          tableName: key,
          indexName: condition.indexName,
          target: condition.target,
          successL: (res) => {
            resolve(res);
          },
        });
      } catch (error) {
        console.log(error);
        reject(false);
      }
    });
  } else {
    return new Promise((resolve, reject) => {
      try {
        const value = store.getItem(key);
        if (value) {
          resolve(JSON.parse(value));
        } else {
          resolve(null);
        }
      } catch (error) {
        reject(false);
      }
    });
  }
}

export function deleteItem(key, store = sessionStorage, condition) {
  if (store.db) {
    return new Promise((resolve, reject) => {
      try {
        store.delete_by_primaryKey({
          tableName: key,
          target: condition.target,
          success: () => {
            resolve(true);
          },
        });
      } catch (error) {
        console.log(error);
        reject(false);
      }
    });
  } else {
    return new Promise((resolve, reject) => {
      try {
        store.deleteItem(key);
        resolve(true);
      } catch (error) {
        reject(false);
      }
    });
  }
}

export function clear(key, store = sessionStorage) {
  if (store.db) {
    return new Promise((resolve, reject) => {
      try {
        store.clear_table({
          tableName: key,
          success: () => {
            resolve(true);
          },
        });
      } catch (error) {
        console.error(error);
        reject(false);
      }
    });
  } else {
    return new Promise((resolve, reject) => {
      try {
        store.clear();
        resolve(true);
      } catch (error) {
        reject(false);
      }
    });
  }
}
