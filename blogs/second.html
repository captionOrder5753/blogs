    # 缓存工具

    前端缓存是性能调优的利器，我们通常把一些定义数据或者不经常变化的数据，根据业务属性存储在不同的介质中，目前支持的介质有：localStorage, sessionStorage，indexDB

    # 参数介绍

    key: 存储的唯一标识 必填
value：存储的值 必填
store: 存储的介质 选填 默认 sessionStorage

    # index.js

    import Idb from "idb.js";

export function getDb(config) {
  return Idb(config);
}

export function setItem(key, value = {}, store = sessionStorage) {
  if (store.db) {
    return new Promise((resolve, reject) => {
      try {
        store.insert({
          tableName: key,
          data: value,
          success: () => resolve(true),
        });
      } catch (error) {
        console.log(error);
        reject(false);
      }
    });
  } else {
    return new Promise((resolve, reject) => {
      try {
        store.setItem(key, JSON.stringify(value));
      } catch (error) {
        console.log(error);
        reject(false);
      }
    });
  }
}

export function getItem(key, store = sessionStorage, condition) {
  if (store.db) {
    return new Promise((resolve, reject) => {
      try {
        store.query_by_index({
          tableName: key,
          indexName: condition.indexName,
          target: condition.target,
          successL: (res) => {
            resolve(res);
          },
        });
      } catch (error) {
        console.log(error);
        reject(false);
      }
    });
  } else {
    return new Promise((resolve, reject) => {
      try {
        const value = store.getItem(key);
        if (value) {
          resolve(JSON.parse(value));
        } else {
          resolve(null);
        }
      } catch (error) {
        reject(false);
      }
    });
  }
}

export function deleteItem(key, store = sessionStorage, condition) {
  if (store.db) {
    return new Promise((resolve, reject) => {
      try {
        store.delete_by_primaryKey({
          tableName: key,
          target: condition.target,
          success: () => {
            resolve(true);
          },
        });
      } catch (error) {
        console.log(error);
        reject(false);
      }
    });
  } else {
    return new Promise((resolve, reject) => {
      try {
        store.deleteItem(key);
        resolve(true);
      } catch (error) {
        reject(false);
      }
    });
  }
}

export function clear(key, store = sessionStorage) {
  if (store.db) {
    return new Promise((resolve, reject) => {
      try {
        store.clear_table({
          tableName: key,
          success: () => {
            resolve(true);
          },
        });
      } catch (error) {
        console.error(error);
        reject(false);
      }
    });
  } else {
    return new Promise((resolve, reject) => {
      try {
        store.clear();
        resolve(true);
      } catch (error) {
        reject(false);
      }
    });
  }
}
